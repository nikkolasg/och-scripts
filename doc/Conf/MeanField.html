def iterate 
    @value.each do |ts,arr|
        ## cannot divide by 0
        arr[1] == 0 ? yield(ts,0) : yield(ts, arr[0]/arr[1])
    end 
end

def save formatted_time,arr
    arr.each do |n|
        next unless @block.call(*arr[1..-1]) if @block
        i = Integer(n) rescue return
        @value[formatted_time][0] += i
        @value[formatted_time][1] += 1
    end
end
